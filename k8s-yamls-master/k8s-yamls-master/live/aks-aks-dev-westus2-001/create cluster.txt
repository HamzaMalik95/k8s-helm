    --name aks-aks-dev-westus2-001 \

    --dns-name-prefix aks-aks-dev-westus2-001 \

    --auto-upgrade-channel patch \
    --node-resource-group rg-akspool-dev-westus2-001 \
    --nodepool-name sysdev01 \
    --vm-set-type VirtualMachineScaleSets \
    --node-vm-size standard_d2s_v5 \
    --generate-ssh-keys \
    --kubelet-config git/k8s/k8s-yamls/live/aks-aks-dev-westus2-001/kubeletconfig.json \
    --zones 1 2 3 \
    --enable-cluster-autoscaler \
    --node-count 3 \
    --max-count 5 \
    --min-count 3 \
    --max-pods 60 \
    --load-balancer-sku standard \
    --outbound-type userAssignedNATGateway \
    --network-plugin azure \
    --network-policy calico \
    --service-cidr 10.247.0.0/18 \
    --dns-service-ip 10.247.0.10 \
    --docker-bridge-address 172.17.0.1/16 \
    --vnet-subnet-id '/subscriptions/6fedb444-f968-49de-b47e-1e02299c1c18/resourceGroups/rg-net-dev-westus2-001/providers/Microsoft.Network/virtualNetworks/vnet-net-dev-westus2-10.225.0.0_16/subnets/snet-aksnodes-dev-westus2-10.225.4.0_24' \

    --uptime-sla \
    --enable-private-cluster \
    --private-dns-zone none \
    --enable-managed-identity \
    --enable-aad \
    --aad-tenant-id 9c02d424-0baa-4499-b49e-a06177b472dc \
